import asyncio
import base64
import json
import os
from typing import Dict, List, Any
from playwright.async_api import async_playwright, Browser, Page
from openai import AsyncOpenAI
from PIL import Image
import io
import time

class VisualBrowserAgent:
    def __init__(self, openai_api_key: str):
        self.client = AsyncOpenAI(api_key=openai_api_key)
        self.browser: Browser = None
        self.page: Page = None
        
    async def start(self, headless: bool = False):
        """Launch browser with stealth settings"""
        playwright = await async_playwright().start()
        self.browser = await playwright.chromium.launch(
            headless=headless,
            args=['--disable-blink-features=AutomationControlled']
        )
        context = await self.browser.new_context(
            viewport={'width': 1920, 'height': 1080},
            user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        )
        self.page = await context.new_page()
        
    async def record_workflow(self, url: str, duration_seconds: int = 60) -> List[Dict]:
        """Record user's navigation for 60 seconds"""
        await self.page.goto(url)
        actions = []
        
        # Inject recording script
        await self.page.evaluate("""
            window.recordedActions = [];
            document.addEventListener('click', (e) => {
                const rect = e.target.getBoundingClientRect();
                window.recordedActions.push({
                    type: 'click',
                    x: rect.x + rect.width/2,
                    y: rect.y + rect.height/2,
                    timestamp: Date.now(),
                    tagName: e.target.tagName,
                    text: e.target.innerText?.substring(0, 100)
                });
            });
        """)
        
        await asyncio.sleep(duration_seconds)
        
        # Extract recorded actions
        actions = await self.page.evaluate("return window.recordedActions || []")
        
        # Capture screenshots of each clicked element
        for action in actions:
            if action['type'] == 'click':
                # Take screenshot of the area around click
                screenshot = await self.page.screenshot(
                    clip={'x': action['x']-100, 'y': action['y']-50, 
                          'width': 200, 'height': 100}
                )
                action['screenshot_base64'] = base64.b64encode(screenshot).decode()
                
        return actions
    
    async def extract_table_data(self, page: Page) -> List[Dict]:
        """Extract data from current table view using GPT-4V"""
        screenshot = await page.screenshot(full_page=True)
        
        # Convert to base64
        img_base64 = base64.b64encode(screenshot).decode()
        
        response = await self.client.chat.completions.create(
            model="gpt-4-vision-preview",
            messages=[
                {
                    "role": "system",
                    "content": "You are a data extraction expert. Extract all visible table rows as JSON array. Include column headers. Return ONLY valid JSON."
                },
                {
                    "role": "user",
                    "content": [
                        {
                            "type": "text",
                            "text": "Extract all rows from this CRM table view. Return as JSON array where each object has keys matching column headers."
                        },
                        {
                            "type": "image_url",
                            "image_url": {"url": f"data:image/png;base64,{img_base64}"}
                        }
                    ]
                }
            ],
            max_tokens=4000,
            temperature=0.1
        )
        
        # Parse JSON response
        content = response.choices[0].message.content
        # Clean markdown code blocks if present
        if '```json' in content:
            content = content.split('```json')[1].split('```')[0]
        elif '```' in content:
            content = content.split('```')[1].split('```')[0]
            
        return json.loads(content)
    
    async def extract_detail_page(self, page: Page) -> Dict:
        """Extract key-value pairs from a detail record page"""
        screenshot = await page.screenshot(full_page=True)
        img_base64 = base64.b64encode(screenshot).decode()
        
        response = await self.client.chat.completions.create(
            model="gpt-4-vision-preview",
            messages=[
                {
                    "role": "system",
                    "content": "Extract all labeled fields from this CRM record page. Return as JSON object with field names as keys."
                },
                {
                    "role": "user",
                    "content": [
                        {
                            "type": "text",
                            "text": "Extract every field label and its corresponding value from this record page. Include section headers if present."
                        },
                        {
                            "type": "image_url",
                            "image_url": {"url": f"data:image/png;base64,{img_base64}"}
                        }
                    ]
                }
            ],
            max_tokens=3000,
            temperature=0.1
        )
        
        content = response.choices[0].message.content
        if '```json' in content:
            content = content.split('```json')[1].split('```')[0]
            
        return json.loads(content)
    
    async def auto_extract_all(
        self, 
        start_url: str, 
        recorded_actions: List[Dict],
        max_records: int = 10000
    ) -> List[Dict]:
        """Autonomously extract all data by replaying recorded workflow"""
        await self.page.goto(start_url)
        await self.page.wait_for_load_state('networkidle')
        
        all_data = []
        page_num = 1
        
        while len(all_data) < max_records:
            print(f"Extracting page {page_num}...")
            
            # Extract current page table data
            try:
                page_data = await self.extract_table_data(self.page)
                all_data.extend(page_data)
                print(f"  Extracted {len(page_data)} records (total: {len(all_data)})")
            except Exception as e:
                print(f"  Extraction failed: {e}")
                break
            
            # Find and click next button using visual pattern matching
            next_button_found = False
            for action in recorded_actions:
                if action.get('type') == 'click' and 'next' in action.get('text', '').lower():
                    # Click at recorded coordinates
                    await self.page.mouse.click(action['x'], action['y'])
                    next_button_found = True
                    await self.page.wait_for_load_state('networkidle')
                    time.sleep(2)  # Human-like delay
                    break
            
            if not next_button_found:
                # Try common pagination selectors
                try:
                    await self.page.click('button:has-text("Next")')
                    await self.page.wait_for_load_state('networkidle')
                    time.sleep(2)
                except:
                    print("No more pages found")
                    break
                    
            page_num += 1
            
            # Checkpoint every 100 records
            if len(all_data) % 100 == 0:
                with open(f'checkpoint_{len(all_data)}.json', 'w') as f:
                    json.dump(all_data, f, indent=2)
                    
        return all_data
    
    async def close(self):
        if self.browser:
            await self.browser.close()

    <script>
        let currentMigrationId = null;
        let recordingInterval = null;
        let recordingSeconds = 60;
        
        async function startMigration() {
            const url = document.getElementById('sourceUrl').value;
            if (!url) {
                alert('Please enter your CRM URL');
                return;
            }
            
            const response = await fetch('/api/migration/start', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ source_url: url, user_id: 'replit_user' })
            });
            
            const data = await response.json();
            currentMigrationId = data.migration_id;
            
            // Show recording card
            document.getElementById('setupCard').classList.add('hidden');
            document.getElementById('recordingCard').classList.remove('hidden');
            
            // Start recording timer
            recordingSeconds = 60;
            startRecordingTimer();
            
            // Open recording URL in new window (would launch browser)
            window.open(data.recording_url, '_blank', 'width=1200,height=800');
        }
        
        function startRecordingTimer() {
            recordingInterval = setInterval(() => {
                recordingSeconds--;
                document.getElementById('timer').textContent = recordingSeconds;
                const progress = ((60 - recordingSeconds) / 60) * 100;
                document.getElementById('recordProgress').style.width = `${progress}%`;
                document.getElementById('recordProgress').textContent = `${Math.round(progress)}%`;
                
                if (recordingSeconds <= 0) {
                    clearInterval(recordingInterval);
                    document.getElementById('completeRecordBtn').disabled = false;
                }
            }, 1000);
        }
        
        async function completeRecording() {
            clearInterval(recordingInterval);
            
            const response = await fetch(`/api/migration/${currentMigrationId}/record`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ duration: 60 })
            });
            
            const data = await response.json();
            
            // Show extraction card
            document.getElementById('recordingCard').classList.add('hidden');
            document.getElementById('extractionCard').classList.remove('hidden');
            
            // Start polling status
            setInterval(checkStatus, 3000);
            checkStatus();
        }
        
        async function checkStatus() {
            if (!currentMigrationId) return;
            
            const response = await fetch(`/api/migration/${currentMigrationId}/status`);
            const status = await response.json();
            
            const progress = (status.extracted_records / status.total_records) * 100;
            document.getElementById('extractProgress').style.width = `${progress}%`;
            document.getElementById('extractProgress').textContent = `${Math.round(progress)}%`;
            
            const statusDiv = document.getElementById('extractStatus');
            statusDiv.innerHTML = `
                <strong>Status:</strong> ${status.status}<br>
                <strong>Records Extracted:</strong> ${status.extracted_records} / ${status.total_records}<br>
                <strong>Started:</strong> ${new Date(status.created_at).toLocaleString()}
            `;
            
            if (status.status === 'completed') {
                document.getElementById('downloadBtn').classList.remove('hidden');
                statusDiv.className = 'status status-success';
                statusDiv.innerHTML += '<br>✅ Migration complete! Click download below.';
            } else if (status.status === 'failed') {
                statusDiv.className = 'status status-error';
            }
        }
        
        async function downloadData() {
            window.location.href = `/api/migration/${currentMigrationId}/download`;
        }

How It Works (The User Experience)
User enters CRM URL → System launches visible browser

User navigates naturally for 60 seconds → AI records clicks, scrolls, pagination patterns

System replays workflow autonomously → GPT-4V extracts data from every screen

Data saved as Parquet → Downloadable, portable, vendor-neutral format

Limitations (Realistic for Production)
Speed: ~1000 records/hour (better than manual, slower than API)

Cost: ~$0.10 per 1000 records (GPT-4V vision API)

Accuracy: ~95% for clean UIs, lower for complex nested tables

Legal: Use only for your own data; may violate ToS of some vendors
