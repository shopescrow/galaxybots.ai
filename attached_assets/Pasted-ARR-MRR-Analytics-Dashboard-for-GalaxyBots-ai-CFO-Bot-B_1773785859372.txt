class ARR_MRR_Dashboard:
    def __init__(self, subscription_data):
        self.subscriptions = subscription_data
        self.current_date = datetime.now()
    
    def calculate_mrr(self):
        """Calculate current MRR"""
        return {
            'new_mrr': self._sum_new_subscriptions(),
            'expansion_mrr': self._sum_upgrades(),
            'contraction_mrr': self._sum_downgrades(),
            'churned_mrr': self._sum_churn(),
            'net_new_mrr': self._calculate_net_mrr(),
            'total_mrr': self._calculate_total_mrr()
        }
    
    def calculate_arr(self):
        """Calculate ARR from MRR"""
        mrr_data = self.calculate_mrr()
        return {
            'current_arr': mrr_data['total_mrr'] * 12,
            'arr_breakdown': {
                'new_customers': mrr_data['new_mrr'] * 12,
                'expansion': mrr_data['expansion_mrr'] * 12,
                'contraction': mrr_data['contraction_mrr'] * 12,
                'churn': mrr_data['churned_mrr'] * 12
            }
        }
    
    def projected_arr(self, months=12):
        """Generate ARR projections"""
        historical = self._get_historical_trends()
        return self._forecast_revenue(historical, months)

class DebtToIncomeAnalyzer:
    """
    Comprehensive DTI analysis tool for SaaS businesses
    Tracks both traditional DTI and SaaS-optimized versions
    """
    
    def __init__(self, financial_data):
        self.financials = financial_data
        self.current_date = datetime.now()
    
    def calculate_dti_ratios(self):
        """Calculate multiple DTI variations for SaaS context"""
        return {
            'traditional_dti': self._traditional_dti(),
            'saas_optimized_dti': self._saas_dti(),
            'forward_looking_dti': self._forward_dti(),
            'debt_service_coverage': self._dscr()
        }
    
    def _traditional_dti(self):
        """Standard DTI: Total Monthly Debt / Gross Monthly Income"""
        return {
            'ratio': self.total_monthly_debt / self.gross_monthly_revenue,
            'interpretation': self._interpret_dti(),
            'risk_level': self._assess_risk()
        }
    
    def _saas_dti(self):
        """
        SaaS-optimized DTI: (Total Debt) / (ARR - Operating Expenses)
        More accurate for subscription businesses
        """
        annual_debt_service = self.total_debt_obligations * self.avg_interest_rate
        available_income = self.arr - self.annual_opex
        
        return {
            'ratio': annual_debt_service / available_income,
            'debt_burden': annual_debt_service,
            'available_income': available_income,
            'coverage_months': available_income / annual_debt_service * 12
        }

class DebtCapacityAnalyzer:
    """
    Determine how much additional debt the company can take on
    """
    
    def calculate_debt_capacity(self):
        current_dti = self.dti_analyzer.calculate_dti_ratios()
        
        scenarios = {
            'conservative': {
                'max_additional_debt': self._max_debt_at_ratio(0.35),
                'monthly_payment_impact': self._calculate_payment_impact(0.35),
                'new_cash_requirement': self._required_new_revenue(0.35)
            },
            'moderate': {
                'max_additional_debt': self._max_debt_at_ratio(0.42),
                'monthly_payment_impact': self._calculate_payment_impact(0.42),
                'new_cash_requirement': self._required_new_revenue(0.42)
            },
            'aggressive': {
                'max_additional_debt': self._max_debt_at_ratio(0.50),
                'monthly_payment_impact': self._calculate_payment_impact(0.50),
                'new_cash_requirement': self._required_new_revenue(0.50)
            }
        }
        
        # Add lender perspective
        scenarios['lender_view'] = self._lender_eligibility(scenarios)
        
        return scenarios
    
    def _max_debt_at_ratio(self, target_ratio):
        """Calculate max additional debt to reach target ratio"""
        current_debt = self.total_debt
        current_income = self.annual_income
        
        max_total_debt = target_ratio * current_income
        return max(0, max_total_debt - current_debt)

class DTIEarlyWarning:
    """
    Monitor DTI triggers and send alerts
    """
    
    def check_triggers(self):
        warnings = []
        
        # Define thresholds
        thresholds = {
            'yellow': 0.40,  # Caution zone
            'orange': 0.45,  # High risk
            'red': 0.50      # Critical
        }
        
        current_dti = self.get_current_dti()
        projected_dti = self.get_projected_dti(months=6)
        
        # Current DTI check
        if current_dti > thresholds['yellow']:
            warnings.append({
                'level': '⚠️ YELLOW',
                'message': f'DTI at {current_dti:.2f} - Entering caution zone',
                'action': 'Review expense reduction opportunities'
            })
        
        # Projected DTI check
        if projected_dti > thresholds['orange']:
            warnings.append({
                'level': '🔥 ORANGE',
                'message': f'Projected DTI at {projected_dti:.2f} in 6 months',
                'action': 'Consider refinancing or growth initiatives now'
            })
        
        # Debt maturity concentration
        if self.has_concentrated_maturities():
            warnings.append({
                'level': '⚠️ WARNING',
                'message': 'Debt maturity concentration detected',
                'action': 'Diversify maturity dates through refinancing'
            })
        
        return warnings
