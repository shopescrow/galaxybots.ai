class GalaxyEmailPDFProcessor:
    """
    ENTERPRISE EMAIL INGESTION SYSTEM
    Monitors email accounts, extracts PDFs, triggers workflows automatically
    Based on production-proven patterns[citation:3][citation:5][citation:7]
    """
    
    def __init__(self, config: Dict):
        self.imap_config = config['imap']
        self.smtp_config = config['smtp']
        self.watch_folders = config.get('watch_folders', [])
        self.processed_log = set()
        
    async def monitor_email_inbox(self):
        """
        Continuous monitoring of email inbox for PDF attachments
        Supports Gmail, Outlook, Exchange, IMAP, POP3
        """
        while True:
            try:
                # Connect to IMAP server
                mail = imaplib.IMAP4_SSL(self.imap_config['server'])
                mail.login(self.imap_config['username'], self.imap_config['password'])
                mail.select('INBOX')
                
                # Search for unread emails with PDF attachments
                status, messages = mail.search(None, 'UNSEEN')
                email_ids = messages[0].split()
                
                for email_id in email_ids:
                    status, msg_data = mail.fetch(email_id, '(RFC822)')
                    msg = email.message_from_bytes(msg_data[0][1])
                    
                    # Extract attachments
                    attachments = self._extract_pdf_attachments(msg)
                    
                    if attachments:
                        # Trigger GalaxyPDF processing pipeline
                        for pdf_data in attachments:
                            await self.process_incoming_pdf(
                                pdf_data=pdf_data,
                                metadata={
                                    'from': msg['From'],
                                    'subject': msg['Subject'],
                                    'date': msg['Date'],
                                    'email_id': email_id.decode()
                                }
                            )
                        
                        # Auto-respond with confirmation
                        await self.send_confirmation_email(
                            to=msg['From'],
                            original_subject=msg['Subject'],
                            pdf_count=len(attachments)
                        )
                
                mail.close()
                mail.logout()
                await asyncio.sleep(5)  # Polling interval
                
            except Exception as e:
                print(f"Email monitoring error: {e}")
                await asyncio.sleep(30)
    
    async def process_incoming_pdf(self, pdf_data: bytes, metadata: Dict):
        """
        Route PDF to appropriate GalaxyBot based on email content
        Enables COMPLETE AUTOMATION from email to insight
        """
        # Classify document type from email context
        doc_type = await self.classify_document(
            subject=metadata['subject'],
            sender=metadata['from']
        )
        
        # Route to appropriate bot
        bot_mapping = {
            'invoice': 'FinanceBot',
            'contract': 'LegalBot', 
            'resume': 'HRBot',
            'report': 'AnalyticsBot'
        }
        
        selected_bot = bot_mapping.get(doc_type, 'DefaultBot')
        
        # Trigger GalaxyBots.ai workflow
        result = await self.trigger_bot_workflow(
            bot_name=selected_bot,
            pdf_data=pdf_data,
            email_context=metadata
        )
        
        # Store in document intelligence database
        await self.store_document_intelligence(
            pdf_data=pdf_data,

        'presentation': ['.ppt', '.pptx', '.odp'],

        'email': ['.msg', '.eml', '.pst'],
        'web': ['.html', '.htm', '.mhtml'],
        'archive': ['.zip', '.rar', '.7z']  # Extract then convert
    }
    
    async def batch_convert_folder(self, 
                                   input_folder: str, 
                                   output_format: str = 'pdf',
                                   recursive: bool = True,
                                   maintain_structure: bool = True):
        """
        Convert entire folder hierarchies
        Watch folder for new files (auto-convert)
        Parallel processing with progress tracking
        """
        files = await self.scan_directory(input_folder, recursive)
        
        # Process in parallel batches
        batch_size = 10
        results = []
        
        for i in range(0, len(files), batch_size):
            batch = files[i:i+batch_size]
            batch_results = await asyncio.gather(*[
                self.convert_file(file, output_format) for file in batch
            ])
            results.extend(batch_results)
        
        return results
    
    async def convert_file(self, input_path: str, output_format: str) -> Dict:
        """
        Intelligent conversion with format detection
        Handles complex documents with tables, images, formatting
        """
        mime_type = magic.from_file(input_path, mime=True)

            'document': self._convert_document_to_pdf,
            'spreadsheet': self._convert_spreadsheet_to_pdf,
            'image': self._convert_image_to_pdf,
            'cad': self._convert_cad_to_pdf
        }
        
        converter = converters.get(input_format, self._fallback_conversion)
        
        result = await converter(input_path, output_format)
        
        # Apply AI enhancement (OCR, formatting preservation)
